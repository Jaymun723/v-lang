# The current grammar supported

constant := INTEGER | FLOATING

unary_op := MINUS

double_op := MINUS | ADD | MULT | DIV | MOD

expr :=
  unary_expr expr

unary_expr :=
  OPEN_PARENT expr CLOSE_PARENT |
  constant |
  unary_op unary_expr |
  func_call

func_call := IDENTIFIER OPEN_PARENT expr CLOSE_PARENT

statement := func_call SEMI | NEWLINE

program := statement program | END_OF_FILE
