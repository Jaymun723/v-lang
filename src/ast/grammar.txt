# The current grammar supported

constant := INTEGER | FLOATING

unary_op := MINUS

double_op := MINUS | ADD | MULT | DIV | MOD

expr :=
  unary_expr expr

unary_expr :=
  OPEN_PARENT expr CLOSE_PARENT |
  constant |
  unary_op unary_expr |
  func_call

func_call := IDENTIFIER OPEN_PARENT expr CLOSE_PARENT

block :=
  OPEN_BRACKET statement CLOSE_BRACKET |

else_tail := . | ELSE block | ELSE IF OPEN_PARENT expr CLOSE_PARENT else_tail

statement :=
  func_call SEMI |
  IF OPEN_PARENT expr CLOSE_PARENT block else_tail |
  WHILE OPEN_PARENT expr CLOSE_PARENT block |
  NEWLINE

statement_list := statement | statement statement_list

program := statement_list | END_OF_FILE
